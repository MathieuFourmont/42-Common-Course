// ft_memchr:

#include <string.h>
#include <stdio.h>

int main()
{
	char	s[] = "hello world";
	char	*ptr = ft_memchr(s, 'l', sizeof(s));
	if (ptr != NULL)
	{
		printf("'l' found at position %ld.\n", ptr-s+1);
		printf("search character found: %s\n", ptr);
	}
	else
		printf("search character not found\n");
	return (0);
}

// ft_split:

#include <stdlib.h>
#include <stdio.h>

int	main()
{
	char	*s = "hello.my.name.is";
	char	c = '.';
	char	**splited = ft_split(s, c);
	size_t	i = 0;
	while (i < 4)
	{
		printf("%s\n", splited[i]);
		++i;
	}
}
